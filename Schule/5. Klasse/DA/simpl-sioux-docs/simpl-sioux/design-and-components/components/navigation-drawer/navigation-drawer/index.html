    Navigation Drawer

    Design

    A navigation drawer is used for navigating to other screens and provides additional functionality as app settings and the switch between dark and light theme.

           

    Usage

    This navigation tool visualizes a hierarchically structured set of elements. It is used as a navigation tool to locate individual elements based on their logical hierarchical relationship. The trees are flexible enough to represent even a large number of objects in multiple hierarchy levels.

    Use When

    • a hierarchy of destination items is represented, with two or more levels
    • a list of five or more (even unrelated) top level destination items is provided
    • there are application global settings like theme switch oder language toggle

    Use Alternatively

    • tabs for navigation between groups of content that are related and at the same level of hierarchy.

    Anatomy

    • A sheet containing a list
    • A list of items representing destinations or actions, optionally grouped by horizontal dividers (full width of drawer), if the content is long
    • An optional header area at the top of the list for brand expression (such as an app title or logo), an account switcher, and more
    • The list can contain expandable tree elements to represent a navigation hierarchy:
      • the relative position of elements within the hierarchy helps users locate them easily
      • it is useful to be able to alter the amount of hierarchical detail displayed by expanding and collapsing the tree branches
    • Each list item describing its meaning using a text label and optional icon (placed before text)
    • For label definition see Wording.
    • Tooltips to make text labels readable when they do not fit the available space without scrolling
    • A scrim block is used directly below the drawers sheet, if the navigation drawer is modal to block interaction with the rest of the app
    • In case of navigation tree a breadcrumb must be provided additionally

    Placement

    • Locate the navigation drawer on the left side (on the opposite side of a side sheet),
    • if the app is set to a left-to-right language, or the right side (for right-to-left language)
    • If there is a bottom app bar, it can also be located at the bottom and be opened from there (for use on mobile only)

    Behavior

    • The standard navigation drawer is either permanently on-screen or opened/closed by a navigation menu icon in the toolbar or a specific control handle, represented by an arrow icon At a defined minimum breakpoint of at least 600dp width, a standard drawer should be replaced with a modal drawer (responsive layout)
    • A modal navigation drawer is closed by tapping/clicking outside the sheet
    • The list can be vertically scrolled (within the drawer without impact on the screens content) when the list content extends the drawers height

    import { NgModule } from '@angular/core';
import { HttpClient, HttpClientModule } from '@angular/common/http';

import { SiouxSidepanelModule, SiouxSidenavModule, SiOUXThemeSwitcherModule, ThemeNotifierService,  SiouxLayoutModule,} from '@simpl/sioux-ng';

import { ComponentInjectorService} from '@simpl/sioux-ng';
import { TranslateLoader, TranslateModule } from '@ngx-translate/core';

  return new TranslateHttpLoader(http);
}

    AppComponent,

  ],

    AngularSplitModule,
    AppRoutingModule,

    app.component.ts

    import { AfterViewInit, Component, OnInit, TemplateRef, ViewChild, ViewContainerRef  } from '@angular/core';
import { TranslateService } from '@ngx-translate/core';
import { AppService, ContentHeaderService, SiouxSidepanelComponent } from '@simpl/sioux-ng';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent implements OnInit, AfterViewInit {
  @ViewChild('lside', { static: true }) lside: SiouxSidepanelComponent;
  @ViewChild('rside', { static: true }) rside: SiouxSidepanelComponent;
  @ViewChild('routerTemplate', { read: TemplateRef, static: true }) tpl: TemplateRef<any>;
  @ViewChild('mobileOutlet', { read: ViewContainerRef, static: true }) mobileOutlet;
  @ViewChild('desktopOutlet', { read: ViewContainerRef }) desktopOutlet;
  @ViewChild('splitpane') hss: any;

  public isMobile: boolean;
  public hasRight: boolean;
  public splitAreas = [{ size: 60, default: 60 }, { size: 50, default: 50 }];

  maticons: string[];
  title = 'SiOUX-WebApp';

  private componentAlive$: Subject<void>;

  constructor(translate: TranslateService, public appService: AppService,
    private readonly _contentHeader: ContentHeaderService) {
    translate.setDefaultLang('en');
    translate.use('en');
    this.componentAlive$ = new Subject<void>();  
  }

  public performIconOperation(iconName) {
    console.log(iconName);
  }

  public ngAfterViewInit() {
    if (this.isMobile) {
      this.mobileOutlet.createEmbeddedView(this.tpl);
    } else {
      this.desktopOutlet.createEmbeddedView(this.tpl);
    }

    this.appService.bpW1024$
      .pipe(takeUntil(this.componentAlive$))
      .subscribe({
        next: this.onbpW1024Next.bind(this)
      });

    this.appService.hasRight$
      .pipe(takeUntil(this.componentAlive$))
      .subscribe({
        next: this.onhasRightNext.bind(this)
      });

    this.appService.left$
      .pipe(takeUntil(this.componentAlive$))
      .subscribe({
        next: this.onNextLeft.bind(this)
      });

    this.appService.right$
      .pipe(takeUntil(this.componentAlive$))
      .subscribe({
        next: this.onNextRight.bind(this)
      });

    this.appService.lside = this.lside;
    this.appService.rside = this.rside;
  }

  public onCutterClick(e: { gutterNum: number; sizes: number[] }) {
    if (this.splitAreas[1].size > 0) {
      this.splitAreas[0].size = 100;
      this.splitAreas[1].size = 0;
    } else if (this.splitAreas[1].size === 0) {
      this.splitAreas[0].size = this.splitAreas[0].default;
      this.splitAreas[1].size = this.splitAreas[1].default;
    }
  }

  onbpW1024Next(bpW1024: boolean) {
    if (this.isMobile !== bpW1024) {
      this.appService.closeRight();
      this.changeEmbeddedView(bpW1024);
    }
    this.isMobile = bpW1024;
  }

  onhasRightNext(hasRight: boolean) {
    this.hasRight = hasRight;
  }

  onNextLeft(leftState: boolean) {
    this.panelState(this.lside, leftState);
  }

  onNextRight(rightState: boolean) {
    this.panelState(this.rside, rightState);
  }

  ngOnInit(): void {
    // Pass the Icon's which is to be placed in header
    this.maticons = ['favorite', 'delete'];

    this._contentHeader.pagetitle.title('SiOUX Test Application'); 

  }

  onOpenMenu(menu: any): void {
    console.log(menu);
  }

  onCutterDragEnd(e: { gutterNum: number; sizes: number[] }) {
    this.splitAreas[0].size = e.sizes[0];
    this.splitAreas[1].size = e.sizes[1];
  }

  private panelState(panel: SiouxSidepanelComponent, panelState: boolean) {

    if (panelState) {
      panel.drawer.open();
    } else {
      panel.drawer.close();
    }
  }

  private changeEmbeddedView(state: boolean) {
    this.clearEmbeddedViews();
    this.createEmbeddedView(state);
  }

  private createEmbeddedView(state: boolean) {
    if (state) {
      this.mobileOutlet.createEmbeddedView(this.tpl);
    } else {
      this.desktopOutlet.createEmbeddedView(this.tpl);
    }
  }

  private clearEmbeddedViews() {
    this.clearEmbeddedView(this.desktopOutlet);
    this.clearEmbeddedView(this.mobileOutlet);
  }

  private clearEmbeddedView(view: ViewContainerRef) {
    view.clear();
    view.remove();
  }
}
