    Responsive Design - Sioux Dynamic Layout

    Details

    If you are implementing a responsive application which supports several devices and screen sizes the Sioux Dynamic Layout will be the right for you. Instead of tweaking the existing frontend implementation via css, you can implement for each screen resolution a different view.

    Like this You are not required to use for each screen size the same controls but you can choose those controls which support our customers the best on the corresponding device.

    For instance the view on desktop mode could show a table:

    On a tablet it could show a one-line table without header:

    And on a mobile phone it could show several cards:

    The data from the backend is transmitted from the backend via a faade to the different views. So the data access needs to be implemented only once.

    Code

    Our proposed source code structure for each application view which should be made responsive is the following:

    DynamicTableComponent

    This component needs to be used in the routing or in the page where the views shall be displayed. It contains an array of SiouxDynViewItems which ensure the usage of the correct views.

    dynamic-table.html

    <div style="overflow: auto; height: 100%; width: 100%;">
    <app-sioux-adaptive-panel-layout
        [views]="adaptiveComponentViews">
    </app-sioux-adaptive-panel-layout>
</div>

    dynamic-table.scss

    :host {
  display: contents;
}

    dynamic-table.ts

    import { Component, OnInit } from '@angular/core';
import { SiouxDynViewItem } from '@simpl/sioux-ng';
import { SingleColumnTableComponent } from '../single-column-table/sioux-single-column-table.component';
import { TableComponent } from '../table/table.component';
import { SiouxCardComponent } from '../sioux-card/sioux-card.component';


@Component({
  selector: 'app-dynamic-table',
  templateUrl: './dynamic-table.component.html',
  styleUrls: ['./dynamic-table.component.scss']
})
export class DynamicTableComponent implements OnInit {

  adaptiveComponentViews: SiouxDynViewItem[] = [];

  ngOnInit() {
    this.adaptiveComponentViews = [
      new SiouxDynViewItem(TableComponent, { minwidth: 1024, maxwidth: 1880 }),
      new SiouxDynViewItem(SingleColumnTableComponent, { minwidth: 560, maxwidth: 1023 }),
      new SiouxDynViewItem(SiouxCardComponent, { minwidth: 320, maxwidth: 559 })
    ];
  }

}

    Service

    dynamic-layout.service.ts

    import { Injectable } from '@angular/core';
import { DynamicLayoutDSA } from '../dsa/dynamic-layout.dsa';
import { DynamicLayoutState } from '../states/dynamic-layout.state';

@Injectable()
export class DynamicLayoutService {

  constructor(private siouxLayoutDSA: DynamicLayoutDSA, private dynamicLayoutState: DynamicLayoutState) {
  }

  public loadTableData() {
    this.dynamicLayoutState.addTableData(this.siouxLayoutDSA.loadTableData());
  }
  public updateTableCheck(index: number, checked: boolean) {
    this.siouxLayoutDSA.updateCheckState(index, checked);
  }
}

    State

    dynamic-layout.state.ts

    import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';
import { Element } from '../dbos/dynamic-layout.dbos';

@Injectable()
export class DynamicLayoutState {

    private tableData = new BehaviorSubject<Element[]>([]);

    public getTableData() {
        return this.tableData;
    }

    public addTableData(ELEMENT_DATA: Element[]) {
        this.tableData.next(ELEMENT_DATA);
    }
}

    Middleware

    The facade interacts between the backend and the different views. It gets the data from the backend and provides it to the views.

    dynamic-layout.facade.ts

    import { Injectable } from '@angular/core';
import { DynamicLayoutService } from './core/services/dynamic-layout.service';
import { DynamicLayoutState } from './core/states/dynamic-layout.state';

@Injectable()
export class DynamicLayoutFacade {

  constructor(private dynamicLayoutService: DynamicLayoutService, private dynamicLayoutState: DynamicLayoutState) {
  }
  public getTableData() {
    return this.dynamicLayoutState.getTableData();
  }

  public loadTableData() {
    this.dynamicLayoutService.loadTableData();
  }
  public updateTableCheckState(index: number, checked: boolean) {
    this.dynamicLayoutService.updateTableCheck(index, checked);
  }
}

    single-column-table.component.ts

    import { Component, OnInit, ViewChild } from '@angular/core';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { Element } from '../modules/dynamic-layout/core/dbos/dynamic-layout.dbos';
import { DynamicLayoutFacade } from '../modules/dynamic-layout/dynamic-layout.facade';


@Component({
  selector: 'app-single-column-table',
  templateUrl: './single-column-table.component.html'

})
export class SingleColumnTableComponent implements OnInit {
  tableData: Element[] = [];
  dataSource: any;
  displayedColumns = ['position'];
  constructor(private siouxLayout2Facade: DynamicLayoutFacade) {

  }
  @ViewChild(MatSort)
  sort: MatSort = new MatSort;

  highlight(element: Element) {
    element.highlighted = !element.highlighted;
  }

  ngOnInit() {
    this.siouxLayout2Facade.loadTableData();

    this.siouxLayout2Facade.getTableData().subscribe(val => {
        // subscribing & dealing with asynchronous Observable result from the Core Layer
        this.tableData = val;
      });
      this.dataSource = new MatTableDataSource(this.tableData);
    this.dataSource.sort = this.sort;

    }
  applyFilter(filterValue: string) {
    this.dataSource.filter = filterValue.trim().toLowerCase();
  }
  onCheckboxClick(index: number, checked: boolean) {
    this.siouxLayout2Facade.updateTableCheckState(index - 1, checked);
  }
}
